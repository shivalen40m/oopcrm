CREATE DATABASE DMS_db;

CREATE TABLE users (
                       user_id SERIAL PRIMARY KEY,
                       username VARCHAR(50) UNIQUE NOT NULL,
                       first_name VARCHAR(50) NOT NULL,
                       last_name VARCHAR(50) NOT NULL,
                       password_hash TEXT NOT NULL,
                       role VARCHAR(20) NOT NULL CHECK (role IN ('ADMIN','SALES','SERVICE')),
                       is_deleted BOOLEAN DEFAULT FALSE,
                       created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

CREATE TABLE customers (
                           customer_id SERIAL PRIMARY KEY,
                           first_name VARCHAR(50) NOT NULL,
                           last_name VARCHAR(50) NOT NULL,
                           phone VARCHAR(20),
                           email VARCHAR(100),
                           address TEXT,
                           is_deleted BOOLEAN DEFAULT FALSE,
                           created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

CREATE TABLE vehicles (
                          vehicle_id SERIAL PRIMARY KEY,
                          vin VARCHAR(50) UNIQUE NOT NULL,
                          brand VARCHAR(50) NOT NULL,
                          model VARCHAR(50) NOT NULL,
                          year INT CHECK (year >= 1900),
                          price NUMERIC(12,2) CHECK (price >= 0),
                          mileage INT CHECK (mileage >= 0),
                          status VARCHAR(20) NOT NULL CHECK (status IN ('AVAILABLE','SOLD','RESERVED')),
                          is_active BOOLEAN DEFAULT TRUE,
                          created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

CREATE TABLE sales (
                       sale_id SERIAL PRIMARY KEY,
                       vehicle_id INT NOT NULL,
                       customer_id INT NOT NULL,
                       user_id INT NOT NULL,
                       payment_method VARCHAR(50),
                       final_price NUMERIC(12,2) CHECK (final_price >= 0),
                       sale_date DATE DEFAULT CURRENT_DATE,
                       created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,

                       CONSTRAINT fk_vehicle
                           FOREIGN KEY (vehicle_id) REFERENCES vehicles(vehicle_id),

                       CONSTRAINT fk_customer
                         FOREIGN KEY (customer_id) REFERENCES customers(customer_id),

                       CONSTRAINT fk_user
                         FOREIGN KEY (user_id) REFERENCES users(user_id)
);

CREATE TABLE services (
                          service_id SERIAL PRIMARY KEY,
                          vehicle_id INT NOT NULL,
                          customer_id INT NOT NULL,
                          description TEXT NOT NULL,
                          cost NUMERIC(10,2) CHECK (cost >= 0),
                          service_date DATE NOT NULL,
                          created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,

                          CONSTRAINT fk_vehicle_service
                            FOREIGN KEY (vehicle_id) REFERENCES vehicles(vehicle_id),

                          CONSTRAINT fk_customer_service
                            FOREIGN KEY (customer_id) REFERENCES customers(customer_id)
);
